## 通用模板

```
func search(nums []int, target int) int {
	if len(nums) == 0 {
		return -1
	}
	start := 0
	end := len(nums) - 1
	for start+1 < end {
		mid := start + (end-start)/2
		if nums[mid] == target {
			return mid
		} else if nums[mid] > target {
			end = mid
		} else {
			start = mid
		}
	}
	if nums[start] == target {
		return start
	}
	if nums[end] == target {
		return end
	}
	return -1
}
```

## 查找最左边相等的

```
func searchLeft(nums []int, target int) int {
	res := -1
	if len(nums) == 0 {
		return res
	}
	start := 0
	end := len(nums) - 1  
	for start+1 < end {
		mid := start + (end-start)/2
		if nums[mid] >= target {
			end = mid
		} else {
			start = mid
		}
	}
	if nums[end] == target {
		res = end
	}
    if nums[start] == target {
		res = start
	}
	return res
}
```

			start = mid
		}
	}
	if nums[start] >= target {
		return start
	} else if nums[end] < target {
		return end + 1
	}
	return end
